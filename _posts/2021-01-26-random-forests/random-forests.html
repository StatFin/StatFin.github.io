Dealing with the names

officers <- c(" Capt",' Col',' Don',' Dr',' Jonkheer',' Major',' Rev',' Sir',' the Countess')
full <- full %>%
   mutate(Title_B = gsub('(.*,)|(\\..*)', '', Name))%>%
   mutate(Title = dplyr::case_when(
                      Title_B == " Mlle" ~ " Miss",
                      Title_B == " Ms" ~ " Miss",
                      Title_B == " Mme" ~ " Mrs",
                      Title_B == " Lady" ~ " Miss",
                      Title_B == " Dona" ~ " Miss",
                      Title_B %in% officers ~ "Officer",
                      TRUE ~ Title_B
  ))
full$Title = factor(full$Title)

Modeling with rpart and caret

#creating train and test sets
train.new <- full[1:ltrain,]
ltest <- ltrain+1
test.new <- full[ltest: nrow(full),]
in.train <- createDataPartition(train.new$Survived, p=0.80,list=FALSE)
trainset <- train.new[in.train,]
testset <- train.new[-in.train,]

#parallel programming 
cl <- makePSOCKcluster(8)
registerDoParallel(cl)

#fit the model 
set.seed(1)
model <- rpart(Survived ~ Pclass + Sex + Fare + Embarked + Title + Age,
               data = trainset,
               method = "class")
    
#stop clusters 
stopCluster(cl) 

#plot
rpart.plot(model)
